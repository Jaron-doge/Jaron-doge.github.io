```c++
void List::insertionSort (ListNode* p, int n)
{
    for(int r = 0; r < n; r++) {
        insertAfter(search(p->data, r, p), p->data);
        p = p->succ; 
        remove(p->pred);
    }
}
```

### 选择排序

```c++
void List::selectionSort(ListNode* p, int n)
{
    ListNode* head = p->pred;
    ListNode* tail = p->succ;
    for(int i = 0; i < n; i++)
        tail = tail->succ;
    while(n > 1) {
        ListNode* max = selectMax(head->succ, n);
        insertBefore(tail, remove(max));
        tail = tail->pred;
        n--;
    }
}
```

### 归并排序

```c++
void List::mergeSort(ListNode* p, int n)
{
    if(n < 2)
        return;
    ListNode* q = p;
    int m = n >> 1;
    for(int i = 0; i < m; i++)		// 均分
        q = q->succ;
    mergeSort(p, m);				// 对前后子列表分别排序
    mergeSort(q, n-m);
    merge(p, m, *this, q, n-m);
}

void List::merge(ListNode* p, int n,List L,ListNode* q,int m)
{
    ListNode* pp = p->pred;
    while(m > 0)
        if(n > 0 && p->data <= q->data) {
            if(q == (p = p->succ))
                break;
            n--;
        }
    	else {
            insertBefore(p, L.remove((q = q->succ)->pred));
            m--;
        }
    p = pp->succ;
}
```
